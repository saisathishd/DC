import java.util.*;
public class Hello {

    public static void main(String[] args) {
		Scanner my=new Scanner(System.in);
		String str=my.nextLine();
		char arr[]=str.toCharArray();
		int len=str.length();
		int index=0;
		while(index<len){
		    int cmp=index+1;
		    while(cmp<len){
		        if(Math.abs(arr[index]-arr[cmp])==Math.abs(index-cmp)){
		            System.out.print(arr[index]+""+arr[cmp]);
		        }cmp++;
		    }index++;
		}
	}
}
