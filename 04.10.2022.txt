//Java language..:)
import java.util.*;
class Codechef {

    public static void main(String[] args) {
        Scanner my=new Scanner(System.in);
        int n=my.nextInt();
        int arr[]=new int[n];
        int c=0,t=0;
        for(int i=0;i<n;i++) {
            arr[i]=my.nextInt();
            if(arr[i]%2!=0)
            c++;
        }
        for(int i=0;i<n;i++) {
            if(arr[i]%2!=0)
            c--;
            if(t==c)
            System.out.print(arr[i]+" ");
            else
            System.out.print("-1 ");
            if(arr[i]%2!=0)
            t++;
        }
    }
}
