import java.util.*;
public class Hello {

    public static void main(String[] args) {
        Scanner my=new Scanner(System.in);
        int n=my.nextInt();
        int [][]arr=new int[n][n];
        int c=0;
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                arr[i][j]=my.nextInt();
            }
        }
        for(int i=0;i<n-1;i++){
            for(int j=0;j<n-1;j++){
                if(arr[i][j]==1&&arr[i+1][j]==1&&arr[i][j+1]==1&&arr[i+1][j+1]==1){
                    c++;
                }
            }
        }
        System.out.print(c);
    }
}
